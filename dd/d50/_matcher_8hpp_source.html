166 template <typename Predicate, typename... AdditionalArgs>
167 requires std::is_move_constructible_v<Predicate>
168 && (... && std::is_move_constructible_v<AdditionalArgs>)
169 class PredicateMatcher
170 {
171 private:
172 using storage_t = std::tuple<detail::to_arg_storage_t<AdditionalArgs>...>;
173 template <typename T>
174 using matches_reference_t = typename detail::to_arg_storage_t<T>::matches_reference;
175
176 public:
177 [[nodiscard]]
178 explicit constexpr PredicateMatcher(
179 Predicate predicate,
180 StringViewT fmt,
181 StringViewT invertedFmt,
182 std::tuple<AdditionalArgs...> additionalArgs = {})
183 noexcept(
184 std::is_nothrow_move_constructible_v<Predicate>
185 && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
186 : m_Predicate{std::move(predicate)},
187 m_FormatString{std::move(fmt)},
188 m_InvertedFormatString{std::move(invertedFmt)},
189 m_AdditionalArgs{std::move(additionalArgs)}
190 {
191 }
192
193 template <typename First, typename... Others>
194 requires std::predicate<
195 const Predicate&,
196 First&,
197 Others&...,
198 matches_reference_t<AdditionalArgs>...>
199 [[nodiscard]]
200 constexpr bool matches(
201 First& first,
202 Others&... others) const
203 noexcept(
204 std::is_nothrow_invocable_v<
205 const Predicate&,
206 First&,
207 Others&...,
208 matches_reference_t<AdditionalArgs>...>)
209 {
210 return std::apply(
211 [&, this](auto&... additionalArgs) {
212 return std::invoke(
213 m_Predicate,
214 first,
215 others...,
216 additionalArgs.as_matches_arg()...);
217 },
218 m_AdditionalArgs);
219 }
220
221 [[nodiscard]]
222 constexpr StringT describe() const
223 {
224 return std::apply(
225 [&, this](auto&... additionalArgs) {
226 return format::vformat(
227 m_FormatString,
228 format::make_format_args(
229 std::invoke(
230 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
231 [](auto&& val) noexcept -> const auto& { return val; },
232 additionalArgs.as_describe_arg())...));
233 },
234 m_AdditionalArgs);
235 }
236
237 [[nodiscard]]
238 constexpr auto operator!() const&
239 requires std::is_copy_constructible_v<Predicate>
240 && std::is_copy_constructible_v<storage_t>
241 {
242 return make_inverted(
243 m_Predicate,
244 m_InvertedFormatString,
245 m_FormatString,
246 m_AdditionalArgs);
247 }
248
249 [[nodiscard]]
250 constexpr auto operator!() &&
251 {
252 return make_inverted(
253 std::move(m_Predicate),
254 std::move(m_InvertedFormatString),
255 std::move(m_FormatString),
256 std::move(m_AdditionalArgs));
257 }
258
259 private:
260 [[no_unique_address]] Predicate m_Predicate;
261 StringViewT m_FormatString;
262 StringViewT m_InvertedFormatString;
263 storage_t m_AdditionalArgs;
264
265 template <typename Fn>
266 [[nodiscard]]
267 static constexpr auto make_inverted(
268 Fn&& fn,
269 StringViewT fmt,
270 StringViewT invertedFmt,
271 storage_t tuple)
272 {
273 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
274 return PredicateMatcher<NotFnT, detail::to_arg_storage_t<AdditionalArgs>...>{
275 std::not_fn(std::forward<Fn>(fn)),
276 std::move(fmt),
277 std::move(invertedFmt),
278 std::move(tuple)};
279 }
280 };

454 void check_fp_value(const std::floating_point auto value)
455 {
456 if (std::isnan(value)
457 || std::isinf(value))
458 {
459 throw std::runtime_error{"Value must be not NaN and not infinity."};
460 }
461 }
462
463 void check_fp_epsilon(const std::floating_point auto epsilon)
464 {
465 if (std::isnan(epsilon)
466 || std::isinf(epsilon)
467 || epsilon <= 0.)
468 {
469 throw std::runtime_error{"Epsilon must be not NaN, not infinity and not less or equal 0."};
470 }
471 }
472 }

482 [[nodiscard]]
483 constexpr auto approx_abs(
484 const std::floating_point auto value,
485 const std::floating_point auto epsilon)
486 {
487 detail::check_fp_value(value);
488 detail::check_fp_epsilon(epsilon);
489
490 return PredicateMatcher{
491 [](const std::floating_point auto target, const auto val, const auto eps) {
492 return std::abs(target - val) <= eps;
493 },
494 "is approximately {} +- {}",
495 "is not approximately {} +- {}",

517 [[nodiscard]]
518 constexpr auto approx_rel(
519 const std::floating_point auto value,
520 const std::floating_point auto relEpsilon)
521 {
522 detail::check_fp_value(value);
523 detail::check_fp_epsilon(relEpsilon);
524
525 return PredicateMatcher{
526 [](const std::floating_point auto target, const auto val, const auto rel) {
527 // when target equals +-infinity, that leads to an inf epsilon, which is not very useful
528 if (!std::isinf(target))
529 {
530 const auto absDiff = std::abs(target - val);
531 const auto scaledEpsilon = rel * std::max(std::abs(target), std::abs(val));
532 return absDiff <= scaledEpsilon;
533 }
534
535 return false;
536 },
537 "is approximately {} +- ({} * max(|lhs|, |rhs|))",
538 "is not approximately {} +- ({} * max(|lhs|, |rhs|))",
539 std::make_tuple(value, relEpsilon)};
540 }
