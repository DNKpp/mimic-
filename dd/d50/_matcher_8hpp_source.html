132 template <typename Predicate, typename... AdditionalArgs>
133 requires std::is_move_constructible_v<Predicate>
134 && (... && std::is_move_constructible_v<AdditionalArgs>)
135 class PredicateMatcher
136 {
137 public:
138 [[nodiscard]]
139 explicit constexpr PredicateMatcher(
140 Predicate predicate,
141 StringT fmt,
142 StringT invertedFmt,
143 std::tuple<AdditionalArgs...> additionalArgs = std::tuple{}
144 ) noexcept(std::is_nothrow_move_constructible_v<Predicate>
145 && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
146 : m_Predicate{std::move(predicate)},
147 m_FormatString{std::move(fmt)},
148 m_InvertedFormatString{std::move(invertedFmt)},
149 m_AdditionalArgs{std::move(additionalArgs)}
150 {
151 }
152
153 template <typename T>
154 requires std::predicate<const Predicate&, T&, const AdditionalArgs&...>
155 [[nodiscard]]
156 constexpr bool matches(
157 T& target
158 ) const noexcept(std::is_nothrow_invocable_v<const Predicate&, T&, const AdditionalArgs&...>)
159 {
160 return std::apply(
161 [&, this](auto&... additionalArgs)
162 {
163 return std::invoke(
164 m_Predicate,
165 target,
166 additionalArgs...);
167 },
168 m_AdditionalArgs);
169 }
170
171 [[nodiscard]]
172 constexpr StringT describe() const
173 {
174 return std::apply(
175 [&, this](auto&... additionalArgs)
176 {
177 return format::vformat(
178 m_FormatString,
179 format::make_format_args(
180 std::invoke(
181 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
182 [](auto&& val) noexcept -> const auto& { return val; },
183 mimicpp::print(additionalArgs))
184 ...));
185 },
186 m_AdditionalArgs);
187 }
188
189 [[nodiscard]]
190 constexpr auto operator !() const &
191 requires std::is_copy_constructible_v<Predicate>
192 && (... && std::is_copy_constructible_v<AdditionalArgs>)
193 {
194 return make_inverted(
195 m_Predicate,
196 m_InvertedFormatString,
197 m_FormatString,
198 std::move(m_AdditionalArgs));
199 }
200
201 [[nodiscard]]
202 constexpr auto operator !() &&
203 {
204 return make_inverted(
205 std::move(m_Predicate),
206 std::move(m_InvertedFormatString),
207 std::move(m_FormatString),
208 std::move(m_AdditionalArgs));
209 }
210
211 private:
212 [[no_unique_address]] Predicate m_Predicate;
213 StringT m_FormatString;
214 StringT m_InvertedFormatString;
215 mutable std::tuple<AdditionalArgs...> m_AdditionalArgs{};
216
217 template <typename Fn>
218 [[nodiscard]]
219 static constexpr auto make_inverted(
220 Fn&& fn,
221 StringT fmt,
222 StringT invertedFmt,
223 std::tuple<AdditionalArgs...> tuple
224 )
225 {
226 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
227 return PredicateMatcher<NotFnT, AdditionalArgs...>{
228 std::not_fn(std::forward<Fn>(fn)),
229 std::move(fmt),
230 std::move(invertedFmt),
231 std::move(tuple)
232 };
233 }
234 };
