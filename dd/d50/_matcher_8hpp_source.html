162 template <typename Predicate, typename... AdditionalArgs>
163 requires std::is_move_constructible_v<Predicate>
164 && (... && std::is_move_constructible_v<AdditionalArgs>)
165 class PredicateMatcher
166 {
167 private:
168 using storage_t = std::tuple<detail::to_arg_storage_t<AdditionalArgs>...>;
169 template <typename T>
170 using matches_reference_t = typename detail::to_arg_storage_t<T>::matches_reference;
171
172 public:
173 [[nodiscard]]
174 explicit constexpr PredicateMatcher(
175 Predicate predicate,
176 StringT fmt,
177 StringT invertedFmt,
178 std::tuple<AdditionalArgs...> additionalArgs = {}) noexcept(std::is_nothrow_move_constructible_v<Predicate> && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
179 : m_Predicate{std::move(predicate)},
180 m_FormatString{std::move(fmt)},
181 m_InvertedFormatString{std::move(invertedFmt)},
182 m_AdditionalArgs{std::move(additionalArgs)}
183 {
184 }
185
186 template <typename T>
187 requires std::predicate<
188 const Predicate&,
189 T&,
190 matches_reference_t<AdditionalArgs>...>
191 [[nodiscard]]
192 constexpr bool matches(
193 T& target) const noexcept(std::is_nothrow_invocable_v<const Predicate&, T&, matches_reference_t<AdditionalArgs>...>)
194 {
195 return std::apply(
196 [&, this](auto&... additionalArgs) {
197 return std::invoke(
198 m_Predicate,
199 target,
200 additionalArgs.as_matches_arg()...);
201 },
202 m_AdditionalArgs);
203 }
204
205 [[nodiscard]]
206 constexpr StringT describe() const
207 {
208 return std::apply(
209 [&, this](auto&... additionalArgs) {
210 return format::vformat(
211 m_FormatString,
212 format::make_format_args(
213 std::invoke(
214 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
215 [](auto&& val) noexcept -> const auto& { return val; },
216 additionalArgs.as_describe_arg())...));
217 },
218 m_AdditionalArgs);
219 }
220
221 [[nodiscard]]
222 constexpr auto operator!() const&
223 requires std::is_copy_constructible_v<Predicate>
224 && std::is_copy_constructible_v<storage_t>
225 {
226 return make_inverted(
227 m_Predicate,
228 m_InvertedFormatString,
229 m_FormatString,
230 m_AdditionalArgs);
231 }
232
233 [[nodiscard]]
234 constexpr auto operator!() &&
235 {
236 return make_inverted(
237 std::move(m_Predicate),
238 std::move(m_InvertedFormatString),
239 std::move(m_FormatString),
240 std::move(m_AdditionalArgs));
241 }
242
243 private:
244 [[no_unique_address]] Predicate m_Predicate;
245 StringT m_FormatString;
246 StringT m_InvertedFormatString;
247 storage_t m_AdditionalArgs;
248
249 template <typename Fn>
250 [[nodiscard]]
251 static constexpr auto make_inverted(
252 Fn&& fn,
253 StringT fmt,
254 StringT invertedFmt,
255 storage_t tuple)
256 {
257 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
258 return PredicateMatcher<NotFnT, detail::to_arg_storage_t<AdditionalArgs>...>{
259 std::not_fn(std::forward<Fn>(fn)),
260 std::move(fmt),
261 std::move(invertedFmt),
262 std::move(tuple)};
263 }
264 };
