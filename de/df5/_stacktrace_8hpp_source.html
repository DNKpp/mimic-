100 concept backend =
101 std::copyable<T>
102 && requires(backend_traits<std::remove_cvref_t<T>> traits, const std::remove_cvref_t<T>& backend, const std::size_t value) {
103 { decltype(traits)::current(value) } -> std::convertible_to<std::remove_cvref_t<T>>;
104 { decltype(traits)::size(backend) } -> std::convertible_to<std::size_t>;
105 { decltype(traits)::empty(backend) } -> std::convertible_to<bool>;

168 class Stacktrace
169 {
170 public:
171 using size_fn = std::size_t (*)(const std::any&);
172 using empty_fn = bool (*)(const std::any&);
173 using description_fn = std::string (*)(const std::any&, std::size_t);
174 using source_file_fn = std::string (*)(const std::any&, std::size_t);
175 using source_line_fn = std::size_t (*)(const std::any&, std::size_t);
176
180 ~Stacktrace() = default;
181
188 template <typename Inner>
189 requires(!std::same_as<Stacktrace, std::remove_cvref_t<Inner>>)
190 && stacktrace::backend<Inner>
191 [[nodiscard]]
192 explicit constexpr Stacktrace(Inner&& inner)
193 : m_Inner{std::forward<Inner>(inner)},
194 m_SizeFn{&stacktrace::detail::size<std::remove_cvref_t<Inner>>},
195 m_EmptyFn{&stacktrace::detail::empty<std::remove_cvref_t<Inner>>},
196 m_DescriptionFn{&stacktrace::detail::description<std::remove_cvref_t<Inner>>},
197 m_SourceFileFn{&stacktrace::detail::source_file<std::remove_cvref_t<Inner>>},
198 m_SourceLineFn{&stacktrace::detail::source_line<std::remove_cvref_t<Inner>>}
199 {
200 }
201
205 [[nodiscard]]
206 Stacktrace(const Stacktrace&) = default;
207
211 Stacktrace& operator=(const Stacktrace&) = default;
212
216 [[nodiscard]]
217 Stacktrace(Stacktrace&&) = default;
218
222 Stacktrace& operator=(Stacktrace&&) = default;
223
228 [[nodiscard]]
229 constexpr std::size_t size() const
230 {
231 return std::invoke(m_SizeFn, m_Inner);
232 }
233
238 [[nodiscard]]
239 constexpr bool empty() const
240 {
241 return std::invoke(m_EmptyFn, m_Inner);
242 }
243
249 [[nodiscard]]
250 constexpr std::string description(const std::size_t at) const
251 {
252 return std::invoke(m_DescriptionFn, m_Inner, at);
253 }
254
260 [[nodiscard]]
261 constexpr std::string source_file(const std::size_t at) const
262 {
263 return std::invoke(m_SourceFileFn, m_Inner, at);
264 }
265
271 [[nodiscard]]
272 constexpr std::size_t source_line(const std::size_t at) const
273 {
274 return std::invoke(m_SourceLineFn, m_Inner, at);
275 }
276
277 [[nodiscard]]
278 friend bool operator==(const Stacktrace& lhs, const Stacktrace& rhs)
279 {
280 return lhs.size() == rhs.size()
281 && std::ranges::all_of(
282 std::views::iota(0u, lhs.size()),
283 [&](const std::size_t index) {
284 return lhs.description(index) == rhs.description(index)
285 && lhs.source_file(index) == rhs.source_file(index)
286 && lhs.source_line(index) == rhs.source_line(index);
287 });
288 }
289
290 private:
291 std::any m_Inner;
292 size_fn m_SizeFn;
293 empty_fn m_EmptyFn;
294 description_fn m_DescriptionFn;
295 source_file_fn m_SourceFileFn;
296 source_line_fn m_SourceLineFn;
297 };

373template <>
374class mimicpp::detail::Printer<mimicpp::Stacktrace>
375{
376public:
377 template <print_iterator OutIter>
378 static OutIter print(OutIter out, const Stacktrace& stacktrace)
379 {
380 if (stacktrace.empty())
381 {
382 return format::format_to(
383 std::move(out),
384 "empty");
385 }
386
387 for (const std::size_t i : std::views::iota(0u, stacktrace.size()))
388 {
389 out = format::format_to(
390 std::move(out),
391 "{} [{}], {}\n",
392 stacktrace.source_file(i),
393 stacktrace.source_line(i),
394 stacktrace.description(i));
395 }
396
397 return out;
398 }
399};
