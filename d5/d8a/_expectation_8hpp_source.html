30 template <typename Return, typename... Params, typename Signature>
31 std::optional<MatchReport> make_match_report(
32 const call::Info<Return, Params...>& call,
33 const Expectation<Signature>& expectation) noexcept
34 {
35 try
36 {
37 return expectation.matches(call);
38 }
39 catch (...)
40 {
41 report_unhandled_exception(
42 make_call_report(call),
43 expectation.report(),
44 std::current_exception());
45 }
46
47 return std::nullopt;
48 }
49
50 template <typename Signature>
51 constexpr auto pick_best_match(std::vector<std::tuple<Expectation<Signature>&, MatchReport>>& matches)
52 {
53 constexpr auto ratings = [](const auto& el) noexcept -> const auto& {
54 return std::get<state_applicable>(
55 std::get<MatchReport>(el).controlReport)
56 .sequenceRatings;
57 };
58
59 auto best = std::ranges::begin(matches);
60 for (auto iter = best + 1;
61 iter != std::ranges::end(matches);
62 ++iter)
63 {
64 if (!sequence::detail::has_better_rating(
65 ratings(*best),
66 ratings(*iter)))
67 {
68 best = iter;
69 }
70 }
71
72 return best;
73 }

267 void push(std::shared_ptr<ExpectationT> expectation)
268 {
269 const std::scoped_lock lock{m_ExpectationsMx};
270
271 assert(
272 std::ranges::find(m_Expectations, expectation) == std::ranges::end(m_Expectations)
273 && "Expectation already belongs to this storage.");
274
275 m_Expectations.emplace_back(std::move(expectation));
276 }
277
285 void remove(std::shared_ptr<ExpectationT> expectation)
286 {
287 const std::scoped_lock lock{m_ExpectationsMx};
288
289 auto iter = std::ranges::find(m_Expectations, expectation);
290 assert(iter != std::ranges::end(m_Expectations) && "Expectation does not belong to this storage.");
291 m_Expectations.erase(iter);
292
293 if (!expectation->is_satisfied())
294 {
295 detail::report_unfulfilled_expectation(
296 expectation->report());
297 }
298 }
299
309 [[nodiscard]]
310 ReturnT handle_call(CallInfoT call)
311 {
312 std::vector<std::tuple<ExpectationT&, MatchReport>> matches{};
313 std::vector<MatchReport> noMatches{};
314 std::vector<MatchReport> inapplicableMatches{};
315
316 for (const std::scoped_lock lock{m_ExpectationsMx};
317 auto& exp : m_Expectations | std::views::reverse)
318 {
319 if (std::optional matchReport = detail::make_match_report(call, *exp))
320 {
321 switch (evaluate_match_report(*matchReport))
322 {
323 using enum MatchResult;
324 case none:
325 noMatches.emplace_back(*std::move(matchReport));
326 break;
327 case inapplicable:
328 inapplicableMatches.emplace_back(*std::move(matchReport));
329 break;
330 case full:
331 matches.emplace_back(*exp, *std::move(matchReport));
332 break;
333 // GCOVR_EXCL_START
334 default:
335 unreachable();
336 // GCOVR_EXCL_STOP
337 }
338 }
339 }
340
341 if (!std::ranges::empty(matches))
342 {
343 auto&& [exp, report] = *detail::pick_best_match(matches);
344
345 // Todo: Avoid the call copy
346 // Maybe we can prevent the copy here, but we should keep the instruction order as-is, because
347 // in cases of a throwing finalizer, we might introduce bugs. At least there are some tests, which
348 // will fail if done wrong.
349 detail::report_full_match(
350 make_call_report(call),
351 std::move(report));
352 exp.consume(call);
353 return exp.finalize_call(call);
354 }
355
356 if (!std::ranges::empty(inapplicableMatches))
357 {
358 detail::report_inapplicable_matches(
359 make_call_report(std::move(call)),
360 std::move(inapplicableMatches));
361 }
362
363 detail::report_no_matches(
364 make_call_report(std::move(call)),
365 std::move(noMatches));
366 }
367
368 private:
369 std::vector<std::shared_ptr<ExpectationT>> m_Expectations{};
370 std::mutex m_ExpectationsMx{};
371 };

376 template <typename T, typename Signature>
377 concept expectation_policy_for = std::is_move_constructible_v<T>
378 && std::is_destructible_v<T>
379 && std::same_as<T, std::remove_cvref_t<T>>
380 && requires(T& policy, const call::info_for_signature_t<Signature>& info) {
381 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
382 { std::as_const(policy).matches(info) } -> std::convertible_to<bool>;
383 { std::as_const(policy).describe() } -> std::convertible_to<std::optional<StringT>>;
384 { policy.consume(info) };
385 };

428 public:
429 using ControlPolicyT = ControlPolicy;

445 template <typename ControlPolicyArg, typename FinalizerArg, typename... PolicyArgs>
446 requires std::constructible_from<ControlPolicyT, ControlPolicyArg>
447 && std::constructible_from<FinalizerT, FinalizerArg>
448 && std::constructible_from<PolicyListT, PolicyArgs...>
449 constexpr explicit BasicExpectation(
450 expectation_info info,
451 ControlPolicyArg&& controlArg,
452 FinalizerArg&& finalizerArg,
453 PolicyArgs&&... args)
454 noexcept(
455 std::is_nothrow_constructible_v<ControlPolicyT, ControlPolicyArg>
456 && std::is_nothrow_constructible_v<FinalizerT, FinalizerArg>
457 && (std::is_nothrow_constructible_v<Policies, PolicyArgs> && ...))
458 : m_Info{std::move(info)},
459 m_ControlPolicy{std::forward<ControlPolicyArg>(controlArg)},
460 m_Policies{std::forward<PolicyArgs>(args)...},
461 m_Finalizer{std::forward<FinalizerArg>(finalizerArg)}
462 {
463 }

468 [[nodiscard]]
469 ExpectationReport report() const override
470 {
471 return ExpectationReport{
472 .expectationInfo = m_Info,
473 .finalizerDescription = std::nullopt,
474 .timesDescription = describe_times(),
475 .expectationDescriptions = std::apply(
476 [&](const auto&... policies) {
477 return std::vector<std::optional<StringT>>{
478 policies.describe()...};
479 },
480 m_Policies)};
481 }
482
486 [[nodiscard]]
487 constexpr bool is_satisfied() const noexcept override
488 {
489 return m_ControlPolicy.is_satisfied()
490 && std::apply(
491 [](const auto&... policies) noexcept {
492 return (... && policies.is_satisfied());
493 },
494 m_Policies);
495 }
496
500 [[nodiscard]]
501 MatchReport matches(const CallInfoT& call) const override
502 {
503 return MatchReport{
504 .expectationInfo = m_Info,
505 .finalizeReport = {std::nullopt},
506 .controlReport = m_ControlPolicy.state(),
507 .expectationReports = gather_expectation_reports(call)};
508 }
509
513 constexpr void consume(const CallInfoT& call) override
514 {
515 m_ControlPolicy.consume();
516 std::apply(
517 [&](auto&... policies) noexcept {
518 (..., policies.consume(call));
519 },
520 m_Policies);
521 }
522
526 [[nodiscard]]
527 constexpr ReturnT finalize_call(const CallInfoT& call) override
528 {
529 return m_Finalizer.finalize_call(call);
530 }

535 [[nodiscard]]
536 constexpr const std::source_location& from() const noexcept override
537 {
538 return m_Info.sourceLocation;
539 }
540
544 [[nodiscard]]
545 constexpr const StringT& mock_name() const noexcept override
546 {
547 return m_Info.mockName;
548 }

550 private:
551 expectation_info m_Info;
552 ControlPolicyT m_ControlPolicy;
553 PolicyListT m_Policies;

556 [[nodiscard]]
557 StringT describe_times() const
558 {
559 StringStreamT ss{};
560 std::visit(
561 std::bind_front(

563 std::ostreambuf_iterator{ss}),
