445 template <typename Signature, typename Mock, typename... Args>
446 constexpr signature_return_type_t<Signature> indirectly_apply_mock(Mock& mock, std::tuple<Args...>&& args)
447 {
448 using mock_ref_t = std::conditional_t<
449 ValueCategory::rvalue == signature_ref_qualification_v<Signature>,
450 Mock&&,
451 Mock&>;
452 const auto forward_apply = [&]<std::size_t... indices>([[maybe_unused]] const std::index_sequence<indices...>)
453 -> signature_return_type_t<Signature> {
454 return static_cast<mock_ref_t>(mock)(
455 std::forward<Args>(std::get<indices>(args))...);
456 };
457
458 constexpr auto sequence = std::index_sequence_for<Args...>{};
459 return forward_apply(sequence);
460 };
461}
