81 template <print_iterator OutIter>
82 OutIter print_times_state(OutIter out, const std::size_t current, const std::size_t min, const std::size_t max)
83 {
84 const auto verbalizeValue = [](OutIter o, const std::size_t value) {
85 switch (value)
86 {
87 case 0: return format::format_to(std::move(o), "never");
88 case 1: return format::format_to(std::move(o), "once");
89 case 2: return format::format_to(std::move(o), "twice");
90 default: return format::format_to(std::move(o), "{} times", value);
91 }
92 };
93
94 if (current == max)
95 {
96 out = format::format_to(
97 std::move(out),
98 "already saturated (matched ");
99 out = verbalizeValue(std::move(out), current);
100 return format::format_to(std::move(out), ")");
101 }
102
103 if (min <= current)
104 {
105 return format::format_to(
106 std::move(out),
107 "accepts further matches (matched {} out of {} times)",
108 current,
109 max);
110 }
111
112 const auto verbalizeInterval = [verbalizeValue](OutIter o, const std::size_t start, const std::size_t end) {
113 if (start < end)
114 {
115 return format::format_to(
116 std::move(o),
117 "between {} and {} times",
118 start,
119 end);
120 }
121
122 o = format::format_to(std::move(o), "exactly ");
123 return verbalizeValue(std::move(o), end);
124 };
125
126 out = format::format_to(std::move(out), "matched ");
127 out = verbalizeValue(std::move(out), current);
128 out = format::format_to(std::move(out), " - ");
129 out = verbalizeInterval(std::move(out), min, max);
130 return format::format_to(std::move(out), " is expected");
131 }

133 struct control_state_printer
134 {
135 template <print_iterator OutIter>
136 OutIter operator()(OutIter out, const state_applicable& state) const
137 {
138 out = print_times_state(
139 std::move(out),
140 state.count,
141 state.min,
142 state.max);
143
144 if (const auto sequenceCount = std::ranges::ssize(state.sequenceRatings);
145 0 < sequenceCount)
146 {
147 out = format::format_to(
148 std::move(out),
149 ",\n\tand is the current element of {} sequence(s).",
150 sequenceCount);
151 }
152
153 return out;
154 }
155
156 template <print_iterator OutIter>
157 OutIter operator()(OutIter out, const state_inapplicable& state) const
158 {
159 out = print_times_state(
160 std::move(out),
161 state.count,
162 state.min,
163 state.max);
164
165 const auto totalSequences = std::ranges::ssize(state.sequenceRatings)
166 + std::ranges::ssize(state.inapplicableSequences);
167 return format::format_to(
168 std::move(out),
169 ",\n\tbut is not the current element of {} sequence(s) ({} total).",
170 std::ranges::ssize(state.inapplicableSequences),
171 totalSequences);
172 }
173
174 template <print_iterator OutIter>
175 OutIter operator()(OutIter out, const state_saturated& state) const
176 {
177 out = print_times_state(
178 std::move(out),
179 state.count,

181 state.max);
182
183 if (const auto sequenceCount = std::ranges::ssize(state.sequences);
184 0 < sequenceCount)
185 {
186 out = format::format_to(
187 std::move(out),
188 ",\n\tand is part of {} sequence(s).",

491 [[nodiscard]]
492 inline MatchResult evaluate_match_report(const MatchReport& report) noexcept
493 {
494 if (!std::ranges::all_of(report.expectationReports, &MatchReport::Expectation::isMatching))
495 {
496 return MatchResult::none;
497 }
498
499 if (!std::holds_alternative<state_applicable>(report.controlReport))
500 {
501 return MatchResult::inapplicable;
502 }
503
504 return MatchResult::full;

507 template <>
508 struct printing::detail::state::common_type_printer<MatchReport>
509 {
510 template <print_iterator OutIter>
511 static OutIter print(OutIter out, const MatchReport& report)
512 {
513 std::vector<StringT> matchedExpectationDescriptions{};
514 std::vector<StringT> unmatchedExpectationDescriptions{};
515
516 for (const auto& [isMatching, description] : report.expectationReports)
517 {
518 if (description)
519 {
520 if (isMatching)
521 {
522 matchedExpectationDescriptions.emplace_back(*description);
523 }
524 else
525 {
526 unmatchedExpectationDescriptions.emplace_back(*description);
527 }
528 }
529 }
530
531 bool printReason = false;
532 switch (evaluate_match_report(report))
533 {
534 case MatchResult::full:
535 out = format::format_to(
536 std::move(out),
537 "Matched expectation: {{\n");
538 break;
539
540 case MatchResult::inapplicable:
541 out = format::format_to(
542 std::move(out),
543 "Inapplicable, but otherwise matched expectation: {{\n");
544 printReason = true;
545 break;
546
547 case MatchResult::none:
548 out = format::format_to(
549 std::move(out),
550 "Unmatched expectation: {{\n");
551 break;
552
553 // GCOVR_EXCL_START
554 default: // NOLINT(clang-diagnostic-covered-switch-default)
555 unreachable();
556 // GCOVR_EXCL_STOP
557 }
558
559 out = format::format_to(
560 std::move(out),
561 "mock: {}\n",
562 report.expectationInfo.mockName);
563
564 if (printReason)
565 {
566 out = format::format_to(std::move(out), "reason: ");
567 out = std::visit(
568 std::bind_front(mimicpp::detail::control_state_printer{}, std::move(out)),
569 report.controlReport);
570 out = format::format_to(std::move(out), "\n");
571 }
572
573 out = format::format_to(
574 std::move(out),
575 "from: ");
576 out = mimicpp::print(
577 std::move(out),
578 report.expectationInfo.sourceLocation);
579 out = format::format_to(
580 std::move(out),
581 "\n");
582
583 if (!std::ranges::empty(unmatchedExpectationDescriptions))
584 {
585 out = format::format_to(
586 std::move(out),
587 "failed:\n");
588 for (const auto& desc : unmatchedExpectationDescriptions)
589 {
590 out = format::format_to(
591 std::move(out),
592 "\t{},\n",
593 desc);
594 }
595 }
596
597 if (!std::ranges::empty(matchedExpectationDescriptions))
598 {
599 out = format::format_to(
600 std::move(out),
601 "passed:\n");
602 for (const auto& desc : matchedExpectationDescriptions)
603 {
604 out = format::format_to(
605 std::move(out),
606 "\t{},\n",
607 desc);
608 }
609 }
610
611 return format::format_to(
612 std::move(out),
613 "}}\n");
614 }
