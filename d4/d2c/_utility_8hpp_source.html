231 template <std::default_initializable FillElement, std::size_t n, typename... Elements>
232 requires(sizeof...(Elements) <= n)
233 [[nodiscard]]
234 constexpr auto expand_tuple(std::tuple<Elements...>&& tuple)
235 {
236 // prior to c++23, tuple_cat does not officially support tuple-like types,
237 // thus we transform the generated array manually
238 return std::tuple_cat(
239 std::move(tuple),
240 std::apply(
241 [](auto&&... elements) { return std::make_tuple(std::move(elements)...); },
242 std::array<FillElement, n - sizeof...(Elements)>{}));
243 }
