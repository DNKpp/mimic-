166 [[nodiscard]]
167 constexpr auto expand_tuple(std::tuple<Elements...>&& tuple)
168 {
169 // prior to c++23, tuple_cat does not officially support tuple-like types,
170 // thus we transform the generated array manually
171 return std::tuple_cat(
172 std::move(tuple),
173 std::apply(
174 [](auto&&... elements) { return std::make_tuple(std::move(elements)...); },
175 std::array<FillElement, n - sizeof...(Elements)>{}));
